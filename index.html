-- RankFlow API Integration Example
local HttpService = game:GetService("HttpService")
local Players = game:GetService("Players")

-- CONFIGURATION
local API_URL = "https://rankflow-production-e079.up.railway.app/api/v2/rank/update" 
local SECRET_TOKEN = "YOUR_TOKEN_HERE" -- Get this via /setup
local GROUP_ID = 35119105 
local MIN_RANK = 40

-- API Calling Function
local function callRankApi(targetId, action)
    local data = {
        userId = tostring(targetId),
        action = action
    }

    local success, response = pcall(function()
        return HttpService:RequestAsync({
            Url = API_URL,
            Method = "POST",
            Headers = {
                ["Content-Type"] = "application/json",
                ["Authorization"] = SECRET_TOKEN
            },
            Body = HttpService:JSONEncode(data)
        })
    end)

    if success and response.StatusCode == 200 then
        print("✅ Ranking Successful!")
        return true
    else
        warn("❌ Request Failed: " .. (response and response.StatusCode or "Error"))
    end
    return false
end
